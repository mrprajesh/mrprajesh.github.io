build gpu graph libraries Graphit, GSwitch and SepGraph

On AQUA.

Gpu Graphit G2

module load cmake3.18 module load gcc640 mkdir GraphItG2; cd GraphItG2; git clone –recursive -b gpu-merge https://github.com/GraphIt-DSL/graphit.git . mkdir build; cd build cmake .. make -j 32

cd ../graphit_eval/g2_cgo2021_eval/dataset make small # or just make python3 gen_table7.py small < input

input /lfs/usrhome/phd/cs16d003/install/GraphItG2/graphit_eval/g2_cgo2021_eval/table7_outputs /lfs/usrhome/phd/cs16d003/install/GraphItG2/graphit_eval/g2_cgo2021_eval/../../build /lfs/usrhome/phd/cs16d003/install/GraphItG2/graphit_eval/g2_cgo2021_eval/dataset /lfs/sware/cuda-10.1/bin/nvcc /lfs/sware/gcc640/bin/g++ 0
